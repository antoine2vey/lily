## Implementation
```typescript
export function AchievementCard({ achievement, onPress }: AchievementCardProps) {
  const isUnlocked = achievement.unlockedAt !== null

  return (
    <Pressable onPress={onPress} style={styles.card}>
      <View style={[styles.iconContainer, !isUnlocked && styles.lockedIcon]}>
        <AchievementIcon type={achievement.icon} locked={!isUnlocked} />
        {!isUnlocked && <LockIcon style={styles.lockOverlay} />}
      </View>
      <Text style={[styles.name, !isUnlocked && styles.lockedName]}>
        {achievement.name}
      </Text>
    </Pressable>
  )
}
```
