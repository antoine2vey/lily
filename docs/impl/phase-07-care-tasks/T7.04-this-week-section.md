## Implementation
```typescript
interface ThisWeekSectionProps {
  tasksByDay: Record<string, CareTask[]> // { 'Tomorrow': [...], 'Wednesday': [...] }
}

export function ThisWeekSection({ tasksByDay }: ThisWeekSectionProps) {
  return (
    <View style={styles.section}>
      <Text style={styles.title}>This Week</Text>
      {Object.entries(tasksByDay).map(([day, tasks]) => (
        <CollapsibleDay key={day} day={day} tasks={tasks} />
      ))}
    </View>
  )
}

function CollapsibleDay({ day, tasks }: { day: string; tasks: CareTask[] }) {
  const [expanded, setExpanded] = useState(false)

  return (
    <View>
      <Pressable onPress={() => setExpanded(!expanded)} style={styles.dayHeader}>
        <View style={styles.dayIcon}>
          <CalendarIcon size={20} />
        </View>
        <View style={styles.dayInfo}>
          <Text style={styles.dayName}>{day}</Text>
          <Text style={styles.taskCount}>{tasks.length} task{tasks.length !== 1 ? 's' : ''} scheduled</Text>
        </View>
        <ChevronIcon direction={expanded ? 'up' : 'down'} />
      </Pressable>
      {expanded && tasks.map((task) => (
        <CareTaskCard key={task.id} task={task} compact />
      ))}
    </View>
  )
}
```

## Visual Specs
- Day rows: 56px height, chevron to expand
- Task count: "X tasks scheduled"
- Expanded: shows task cards below
