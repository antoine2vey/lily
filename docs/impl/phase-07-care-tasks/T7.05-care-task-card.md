## Props Interface

```typescript
interface CareTaskCardProps {
  task: {
    id: string
    plantId: string
    plantName: string
    plantImageUrl?: string
    type: 'water' | 'fertilize' | 'prune' | 'rotate' | 'mist'
    completed: boolean
  }
  onPress: () => void
  onComplete: () => void
  overdue?: boolean
  compact?: boolean
}
```

## Implementation
```typescript
const TASK_CONFIG = {
  water: { icon: DropletIcon, color: '#2196F3', label: 'THIRSTY' },
  fertilize: { icon: SparklesIcon, color: '#FF9800', label: 'FERTILIZE' },
  prune: { icon: ScissorsIcon, color: '#E91E63', label: 'PRUNE' },
  rotate: { icon: RotateIcon, color: '#9C27B0', label: 'ROTATE' },
  mist: { icon: CloudIcon, color: '#90CAF9', label: 'MIST' },
}

export function CareTaskCard({ task, onPress, onComplete, overdue, compact }: CareTaskCardProps) {
  const config = TASK_CONFIG[task.type]

  return (
    <Pressable onPress={onPress} style={[styles.card, overdue && styles.overdueCard]}>
      <Image source={{ uri: task.plantImageUrl }} style={styles.plantImage} />
      <View style={styles.content}>
        <Text style={styles.plantName}>{task.plantName}</Text>
        <Badge label={config.label} variant={overdue ? 'error' : 'info'} icon={<config.icon size={12} />} />
      </View>
      <Pressable onPress={onComplete} style={styles.checkButton}>
        {task.completed ? (
          <CheckCircleIcon color="#5B8C5A" size={28} />
        ) : (
          <CircleIcon color="#E0E0E0" size={28} />
        )}
      </Pressable>
    </Pressable>
  )
}
```

## Visual Specs
