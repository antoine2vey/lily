## Implementation
```typescript
const navigation = useNavigation()

const handleTaskPress = (task: CareTask) => {
  navigation.navigate('PlantDetail', { plantId: task.plantId })
}

<CareTaskCard
  task={task}
  onPress={() => handleTaskPress(task)}
  onComplete={() => handleTaskComplete(task.id)}
/>
```
