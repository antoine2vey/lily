## Implementation
```typescript
export function useCompleteTask() {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: async (taskId: string) => {
      await api.careTasks.complete(taskId)
    },
    onMutate: async (taskId) => {
      await queryClient.cancelQueries(['careTasks'])
      const previous = queryClient.getQueryData(['careTasks'])

      // Optimistic update
      queryClient.setQueryData(['careTasks'], (old: CareTask[]) =>
        old.map((task) =>
          task.id === taskId ? { ...task, completed: true } : task
        )
      )

      return { previous }
    },
    onError: (err, taskId, context) => {
      queryClient.setQueryData(['careTasks'], context?.previous)
    },
    onSettled: () => {
      queryClient.invalidateQueries(['careTasks'])
      queryClient.invalidateQueries(['plants'])
    },
  })
}

// In CareScreen
const { mutate: completeTask } = useCompleteTask()

const handleTaskComplete = (taskId: string) => {
  completeTask(taskId)
}
```
