## Implementation
```typescript
interface OverdueSectionProps {
  tasks: CareTask[]
  onTaskPress: (task: CareTask) => void
  onTaskComplete: (taskId: string) => void
}

export function OverdueSection({ tasks, onTaskPress, onTaskComplete }: OverdueSectionProps) {
  if (tasks.length === 0) return null

  return (
    <View style={styles.section}>
      <View style={styles.header}>
        <Text style={styles.title}>Overdue</Text>
        <Badge label={`${tasks.length} Action Required`} variant="error" />
      </View>
      {tasks.map((task) => (
        <CareTaskCard
          key={task.id}
          task={task}
          onPress={() => onTaskPress(task)}
          onComplete={() => onTaskComplete(task.id)}
          overdue
        />
      ))}
    </View>
  )
}
```

## Visual Specs
