## Implementation
```typescript
const [watering, setWatering] = useState(plant.waterNeeds ?? 0.5)
const [light, setLight] = useState(plant.lightNeeds ?? 0.5)
const [humidity, setHumidity] = useState(plant.humidityNeeds ?? 0.5)

<View style={styles.section}>
  <Text style={styles.sectionTitle}>Care Needs</Text>

  <Slider
    icon={<DropletIcon color="#2196F3" />}
    label="Watering"
    value={watering}
    onValueChange={setWatering}
    minLabel="DRY"
    maxLabel="WET"
    valueLabel={getWateringLabel(watering)}
  />

  <Slider
    icon={<SunIcon color="#FFC107" />}
    label="Light"
    value={light}
    onValueChange={setLight}
    minLabel="LOW"
    maxLabel="DIRECT"
    valueLabel={getLightLabel(light)}
  />

  <Slider
    icon={<CloudIcon color="#90CAF9" />}
    label="Humidity"
    value={humidity}
    onValueChange={setHumidity}
    minLabel="LOW"
    maxLabel="TROPICAL"
    valueLabel={getHumidityLabel(humidity)}
  />
</View>
```
