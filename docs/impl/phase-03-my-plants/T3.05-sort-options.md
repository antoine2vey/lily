## Implementation
```typescript
const SORT_OPTIONS = [
  { key: 'name', label: 'Name (A-Z)' },
  { key: 'dateAdded', label: 'Recently Added' },
  { key: 'nextWater', label: 'Needs Water Soon' },
  { key: 'health', label: 'Needs Attention' },
]

export function SortOptionsSheet({ visible, onClose, selectedOption, onSelect }: SortOptionsSheetProps) {
  return (
    <BottomSheet visible={visible} onClose={onClose} title="Sort by">
      {SORT_OPTIONS.map(({ key, label }) => (
        <ListRow
          key={key}
          title={label}
          rightElement={selectedOption === key && <CheckIcon color="#5B8C5A" />}
          onPress={() => {
            onSelect(key)
            onClose()
          }}
        />
      ))}
    </BottomSheet>
  )
}
```
