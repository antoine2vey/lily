export function ViewToggle({ view, onToggle }: ViewToggleProps) {
  const Icon = view === 'grid' ? ListIcon : GridIcon

  return (
    <Pressable onPress={onToggle} style={styles.button}>
      <Icon size={24} color="#1A1A1A" />
    </Pressable>
  )
}
