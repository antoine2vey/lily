## Implementation
```typescript
export function PlantSearchBar({ value, onChangeText, onClear, placeholder = 'Search plants...' }: PlantSearchBarProps) {
  return (
    <View style={styles.container}>
      <SearchIcon size={20} color="#9E9E9E" />
      <TextInput
        style={styles.input}
        value={value}
        onChangeText={onChangeText}
        placeholder={placeholder}
        placeholderTextColor="#9E9E9E"
      />
      {value.length > 0 && (
        <Pressable onPress={onClear}>
          <XCircleIcon size={20} color="#9E9E9E" />
        </Pressable>
      )}
    </View>
  )
}
```

## Visual Specs
