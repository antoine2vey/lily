## Implementation
```typescript
const NOTIFICATION_ICONS = {
  care_reminder: { icon: DropletIcon, color: '#2196F3' },
  achievement: { icon: TrophyIcon, color: '#FFC107' },
  tip: { icon: LightbulbIcon, color: '#FF9800' },
  system: { icon: BellIcon, color: '#9E9E9E' },
}

export function NotificationItem({ notification, onPress }: NotificationItemProps) {
  const config = NOTIFICATION_ICONS[notification.type]

  return (
    <Pressable onPress={() => onPress(notification)} style={styles.container}>
      <View style={[styles.iconContainer, { backgroundColor: `${config.color}20` }]}>
        <config.icon size={20} color={config.color} />
      </View>
      <View style={styles.content}>
        <Text style={styles.title}>{notification.title}</Text>
        <Text style={styles.body}>{notification.body}</Text>
        <Text style={styles.time}>{formatRelativeTime(notification.createdAt)}</Text>
      </View>
      {!notification.read && <View style={styles.unreadDot} />}
    </Pressable>
  )
}
```

## Visual Specs
- Icon: 40px circle, colored background
- Title: 16px, semibold
- Body: 14px, #757575, max 2 lines
- Time: 12px, #9E9E9E
- Unread dot: 8px, #E8997E, right side
