```typescript
const WATERING_PRESETS = [
  { days: 3, label: '3 days' },
  { days: 7, label: '7 days' },
  { days: 14, label: '14 days' },
  { days: 30, label: '30 days' },
]

const FERTILIZING_PRESETS = [
  { days: 14, label: '14 days' },
  { days: 30, label: '30 days' },
  { days: 60, label: '60 days' },
]

export function ManualAddScheduleScreen() {
  const { basicInfo, careNeeds } = useRoute().params
  const [wateringDays, setWateringDays] = useState(7)
  const [fertilizingDays, setFertilizingDays] = useState(30)
  const [careReminders, setCareReminders] = useState(true)
  const [notes, setNotes] = useState('')

  const { mutate: createPlant, isLoading } = useCreatePlant()
  const navigation = useNavigation()

  const handleFinish = () => {
    createPlant({
      ...basicInfo,
      ...careNeeds,
      wateringFrequency: wateringDays,
      fertilizingFrequency: fertilizingDays,
      notificationsEnabled: careReminders,
      notes,
    }, {
      onSuccess: (plant) => {
        navigation.navigate('PlantDetail', { plantId: plant.id })
      },
    })
  }

  return (
    <SafeAreaView style={styles.container}>
      <WizardHeader step={3} totalSteps={3} onBack={navigation.goBack} />

      <ScrollView style={styles.content}>
        <FrequencyPicker
          icon={<DropletIcon />}
          label="Watering Schedule"
          value={wateringDays}
          onValueChange={setWateringDays}
          presets={WATERING_PRESETS}
        />

        <FrequencyPicker
          icon={<SparklesIcon />}
          label="Fertilizing Schedule"
          value={fertilizingDays}
          onValueChange={setFertilizingDays}
          presets={FERTILIZING_PRESETS}
        />

        <ToggleRow
          label="Care Reminders"
          description="Get notified when tasks are due"
          value={careReminders}
          onValueChange={setCareReminders}
        />

        <Input
          label="Notes"
          value={notes}
          onChangeText={setNotes}
          placeholder="Add care tips or specific needs..."
          multiline
          numberOfLines={4}
        />
      </ScrollView>

      <View style={styles.footer}>
        <Button variant="outline" onPress={navigation.goBack}>Back</Button>
        <Button onPress={handleFinish} loading={isLoading}>
          Finish & Add Plant
        </Button>
      </View>
    </SafeAreaView>
  )
}
```
