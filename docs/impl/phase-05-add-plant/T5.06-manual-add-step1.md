## Implementation
```typescript
export function ManualAddBasicInfoScreen() {
  const { prefill } = useRoute().params ?? {}
  const [photo, setPhoto] = useState<string | null>(null)
  const [name, setName] = useState(prefill?.name ?? '')
  const [category, setCategory] = useState(prefill?.category ?? '')

  const navigation = useNavigation()

  const handleNext = () => {
    navigation.navigate('ManualAddCareNeeds', {
      basicInfo: { photo, name, category },
    })
  }

  const canProceed = name.trim().length > 0

  return (
    <SafeAreaView style={styles.container}>
      <WizardHeader step={1} totalSteps={3} onBack={navigation.goBack} />

      <ScrollView style={styles.content}>
        <PhotoPicker
          photo={photo}
          onPickPhoto={setPhoto}
          placeholder="Tap to add photo"
        />

        <Input
          label="Nickname or Plant Name"
          value={name}
          onChangeText={setName}
          placeholder="e.g. Monstera Deliciosa"
          required
        />

        <CategoryPicker
          value={category}
          onSelect={setCategory}
          label="Category"
        />
      </ScrollView>

      <View style={styles.footer}>
        <Button onPress={handleNext} disabled={!canProceed}>
          Next
        </Button>
      </View>
    </SafeAreaView>
  )
}
```
