## Implementation
```typescript
export function AIIdentificationResultsScreen() {
  const { photoUri } = useRoute<AIResultsRoute>().params
  const { data: result, isLoading, error } = useIdentifyPlant(photoUri)

  if (isLoading) return <IdentificationLoading />
  if (error) return <IdentificationError onRetry={retry} />

  return (
    <SafeAreaView style={styles.container}>
      <ScrollView>
        <Image source={{ uri: photoUri }} style={styles.photo} />
        <View style={styles.matchBadge}>
          <Text>{result.confidence}% Match</Text>
        </View>

        <View style={styles.content}>
          <Text style={styles.name}>{result.name}</Text>
          <Text style={styles.commonName}>{result.commonName}</Text>

          <View style={styles.tags}>
            <Chip label={result.category} />
            <Chip label={result.environment} />
          </View>

          <SectionHeader title="Suggested Care" />
          <CareInfoRow icon="water" label="Water" value={result.waterNeeds} />
          <CareInfoRow icon="sun" label="Light" value={result.lightNeeds} />
          <CareInfoRow icon="humidity" label="Humidity" value={result.humidityNeeds} />
        </View>
      </ScrollView>

      <View style={styles.actions}>
        <Button onPress={handleAddToCollection}>Add to Collection</Button>
        <View style={styles.secondaryActions}>
          <TextButton onPress={handleEdit}>Edit Details</TextButton>
          <TextButton onPress={handleRetry}>Try Again</TextButton>
        </View>
      </View>
    </SafeAreaView>
  )
}
```

## Visual Specs
- Photo: full width, aspect 4:3
- Match badge: green pill, absolute positioned on photo
- Name: 24px, bold
- Common name: 16px, #757575
- Care info: icon + label + value rows
