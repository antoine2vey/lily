## Implementation
```typescript
export function AddPlantOptionsSheet({ visible, onClose, onSelectAI, onSelectScan, onSelectManual }: AddPlantOptionsSheetProps) {
  return (
    <BottomSheet visible={visible} onClose={onClose} title="Add Plant">
      <ListRow
        leftIcon={<CameraIcon color="#5B8C5A" />}
        title="Identify with AI"
        subtitle="Snap a photo for instant info"
        showChevron
        onPress={() => { onClose(); onSelectAI(); }}
      />
      <ListRow
        leftIcon={<TagIcon color="#5B8C5A" />}
        title="Scan nursery card"
        subtitle="Read the tag from the store"
        showChevron
        onPress={() => { onClose(); onSelectScan(); }}
      />
      <ListRow
        leftIcon={<EditIcon color="#5B8C5A" />}
        title="Add manually"
        subtitle="Type in the name yourself"
        showChevron
        onPress={() => { onClose(); onSelectManual(); }}
      />
    </BottomSheet>
  )
}
```
