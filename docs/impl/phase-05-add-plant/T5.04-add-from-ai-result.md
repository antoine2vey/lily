## Implementation
```typescript
const { mutate: createPlant, isLoading: isCreating } = useCreatePlant()

const handleAddToCollection = () => {
  createPlant({
    name: result.name,
    species: result.species,
    category: result.category,
    imageUri: photoUri,
    wateringFrequency: result.suggestedWateringDays,
    fertilizingFrequency: result.suggestedFertilizingDays,
    lightNeeds: result.lightNeeds,
    waterNeeds: result.waterNeeds,
    humidityNeeds: result.humidityNeeds,
  }, {
    onSuccess: (plant) => {
      navigation.navigate('PlantDetail', { plantId: plant.id })
    },
  })
}

<Button
  onPress={handleAddToCollection}
  loading={isCreating}
  disabled={isCreating}
>
  Add to Collection
</Button>
```
