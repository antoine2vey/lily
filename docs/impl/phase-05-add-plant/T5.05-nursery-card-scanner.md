## Implementation
```typescript
export function NurseryCardScannerScreen() {
  const [isScanning, setIsScanning] = useState(true)
  const navigation = useNavigation()

  const handleTextRecognized = async (result: TextRecognitionResult) => {
    if (!isScanning) return
    setIsScanning(false)

    // Parse nursery card text
    const plantInfo = parseNurseryCardText(result.text)
    if (plantInfo) {
      navigation.navigate('ManualAddBasicInfo', { prefill: plantInfo })
    } else {
      setIsScanning(true)
    }
  }

  return (
    <View style={styles.container}>
      <Camera
        style={styles.camera}
        onTextRecognized={handleTextRecognized}
      />
      <View style={styles.overlay}>
        <View style={styles.scanArea}>
          <Text style={styles.hint}>Hold steady...</Text>
          <Text style={styles.subHint}>Align tag within frame</Text>
        </View>
      </View>
      <View style={styles.controls}>
        <CloseButton onPress={navigation.goBack} />
        <GalleryButton onPress={handlePickFromGallery} />
      </View>
    </View>
  )
}
```
