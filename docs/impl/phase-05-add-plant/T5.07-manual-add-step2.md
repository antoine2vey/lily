## Implementation
```typescript
export function ManualAddCareNeedsScreen() {
  const { basicInfo } = useRoute().params
  const [watering, setWatering] = useState(0.5)
  const [light, setLight] = useState(0.5)
  const [humidity, setHumidity] = useState(0.5)
  const [petSafe, setPetSafe] = useState(false)

  const navigation = useNavigation()

  const handleNext = () => {
    navigation.navigate('ManualAddSchedule', {
      basicInfo,
      careNeeds: { watering, light, humidity, petSafe },
    })
  }

  return (
    <SafeAreaView style={styles.container}>
      <WizardHeader step={2} totalSteps={3} onBack={navigation.goBack} />

      <ScrollView style={styles.content}>
        <Text style={styles.title}>Care Needs</Text>
        <Text style={styles.subtitle}>Define the ideal environment for your plant.</Text>

        <Slider
          icon={<DropletIcon />}
          label="Watering"
          value={watering}
          onValueChange={setWatering}
          minLabel="Drought tolerant"
          maxLabel="Loves water"
        />

        <Slider
          icon={<SunIcon />}
          label="Light"
          value={light}
          onValueChange={setLight}
          minLabel="Low light"
          maxLabel="Direct sun"
        />

        <Slider
          icon={<CloudIcon />}
          label="Humidity"
          value={humidity}
          onValueChange={setHumidity}
          minLabel="Dry air OK"
          maxLabel="High humidity"
        />

        <ToggleRow
          icon={<PawIcon />}
          label="Pet Safety"
          description="Is this plant toxic to pets?"
          value={petSafe}
          onValueChange={setPetSafe}
        />
      </ScrollView>

      <View style={styles.footer}>
        <Button variant="outline" onPress={navigation.goBack}>Back</Button>
        <Button onPress={handleNext}>Next Step</Button>
      </View>
    </SafeAreaView>
  )
}
```
