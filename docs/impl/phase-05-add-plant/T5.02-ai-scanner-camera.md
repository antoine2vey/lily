## Implementation
```typescript
export function AIScannerScreen() {
  const [permission, requestPermission] = useCameraPermission()
  const cameraRef = useRef<Camera>(null)
  const navigation = useNavigation()

  const handleCapture = async () => {
    if (cameraRef.current) {
      const photo = await cameraRef.current.takePhoto()
      navigation.navigate('AIIdentificationResults', { photoUri: photo.uri })
    }
  }

  if (!permission?.granted) {
    return <CameraPermissionRequest onRequest={requestPermission} />
  }

  return (
    <View style={styles.container}>
      <Camera ref={cameraRef} style={styles.camera} />
      <View style={styles.overlay}>
        <ScanFrame />
        <Text style={styles.hint}>Position your plant in the frame</Text>
      </View>
      <View style={styles.controls}>
        <Pressable onPress={navigation.goBack} style={styles.closeButton}>
          <XIcon color="#FFF" size={24} />
        </Pressable>
        <Pressable onPress={handleCapture} style={styles.captureButton} />
        <Pressable style={styles.flashButton}>
          <ZapIcon color="#FFF" size={24} />
        </Pressable>
      </View>
    </View>
  )
}
```

## Visual Specs
- Full screen camera
- Green corner brackets overlay (scan frame)
- Hint text: white, centered below frame
- Bottom controls: X (close), capture circle, flash toggle
- Capture button: 72px white circle with border
