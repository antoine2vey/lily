## Implementation
```typescript
const THEME_OPTIONS = [
  { key: 'light', label: 'Light', icon: SunIcon },
  { key: 'dark', label: 'Dark', icon: MoonIcon },
  { key: 'system', label: 'System', icon: SmartphoneIcon },
]

export function ThemeSelectionModal({ visible, onClose }: ThemeSelectionModalProps) {
  const { theme, setTheme } = useTheme()

  const handleSelect = async (newTheme: string) => {
    setTheme(newTheme)
    await AsyncStorage.setItem('theme', newTheme)
    onClose()
  }

  return (
    <BottomSheet visible={visible} onClose={onClose} title="Theme">
      {THEME_OPTIONS.map(({ key, label, icon: Icon }) => (
        <ListRow
          key={key}
          leftIcon={<Icon />}
          title={label}
          rightElement={theme === key && <CheckIcon color="#5B8C5A" />}
          onPress={() => handleSelect(key)}
        />
      ))}
      <Button onPress={onClose} style={styles.doneButton}>Done</Button>
    </BottomSheet>
  )
}
```
