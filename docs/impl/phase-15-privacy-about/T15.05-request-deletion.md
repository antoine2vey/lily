## Implementation

```typescript
  <Text style={styles.deletionText}>Request Data Deletion</Text>

const handleRequestDeletion = () => {
  Alert.alert(
    'Request Data Deletion',
    'This will delete all your data and close your account. Are you sure?',
    [
      { text: 'Cancel', style: 'cancel' },
      {
        text: 'Request Deletion',
        style: 'destructive',
        onPress: () => Linking.openURL('mailto:privacy@lily.app?subject=Data Deletion Request'),
      },
    ]
  )
}
```
