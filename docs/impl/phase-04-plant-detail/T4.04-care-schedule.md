## Implementation
```typescript
export function CareSchedule({ wateringDays, wateringDate, fertilizingDays, fertilizingDate, onEdit }: CareScheduleProps) {
  return (
    <View>
      <SectionHeader title="Care Schedule" action={{ label: 'Edit', onPress: onEdit }} />
      <View style={styles.cards}>
        <CareCard
          icon={<DropletIcon color="#2196F3" />}
          days={wateringDays}
          label="Watering"
          nextDate={wateringDate}
        />
        <CareCard
          icon={<SparklesIcon color="#FF9800" />}
          days={fertilizingDays}
          label="Fertilizing"
          nextDate={fertilizingDate}
        />
      </View>
    </View>
  )
}

function CareCard({ icon, days, label, nextDate }) {
  const isOverdue = days < 0
  const displayDays = Math.abs(days)

  return (
    <View style={[styles.card, isOverdue && styles.overdueCard]}>
      <View style={styles.daysCircle}>
        <Text style={styles.daysNumber}>{displayDays}</Text>
        <Text style={styles.daysLabel}>DAYS</Text>
      </View>
      <Text style={styles.cardLabel}>{label}</Text>
      <Text style={styles.nextDate}>{nextDate}</Text>
    </View>
  )
}
```

## Visual Specs

- Days circle: 64px, light green border, green text
- Days number: 24px, bold
- Label: 14px, medium
