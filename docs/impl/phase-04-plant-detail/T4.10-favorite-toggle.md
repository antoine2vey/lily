## Implementation
```typescript
// Hook
export function useFavoritePlant(plantId: string) {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: async (isFavorite: boolean) => {
      await api.plants.updatePlant(plantId, { isFavorite })
    },
    onMutate: async (isFavorite) => {
      // Optimistic update
      await queryClient.cancelQueries(['plant', plantId])
      const previous = queryClient.getQueryData(['plant', plantId])
      queryClient.setQueryData(['plant', plantId], (old) => ({
        ...old,
        isFavorite,
      }))
      return { previous }
    },
    onError: (err, variables, context) => {
      queryClient.setQueryData(['plant', plantId], context.previous)
    },
  })
}

// In PlantDetailScreen
const { mutate: toggleFavorite } = useFavoritePlant(plant.id)

const handleToggleFavorite = () => {
  toggleFavorite(!plant.isFavorite)
}
```
