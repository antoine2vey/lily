## Implementation
```typescript
const ACTIONS = [
  { key: 'water', icon: DropletIcon, label: 'Water' },
  { key: 'fertilize', icon: SparklesIcon, label: 'Fertilize' },
  { key: 'photo', icon: CameraIcon, label: 'Photo' },
  { key: 'chat', icon: MessageSquareIcon, label: 'Chat' },
]

export function QuickActions({ onWater, onFertilize, onPhoto, onChat }: QuickActionsProps) {
  const handlers = { water: onWater, fertilize: onFertilize, photo: onPhoto, chat: onChat }

  return (
    <View style={styles.container}>
      {ACTIONS.map(({ key, icon: Icon, label }) => (
        <Pressable key={key} onPress={handlers[key]} style={styles.action}>
          <View style={styles.iconContainer}>
            <Icon size={24} color="#5B8C5A" />
          </View>
          <Text style={styles.label}>{label}</Text>
        </Pressable>
      ))}
    </View>
  )
}
```

## Visual Specs
- Container: horizontal row, space-around
- Icon container: 48px circle, #E8F5E8 background
- Icon: 24px, #5B8C5A
- Label: 12px, regular, #757575
