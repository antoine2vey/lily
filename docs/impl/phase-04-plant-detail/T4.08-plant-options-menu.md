## Implementation
```typescript
export function PlantOptionsSheet({
  visible, onClose, plantName, isFavorite,
  onEdit, onToggleFavorite, onExportHistory, onShare, onDelete
}: PlantOptionsSheetProps) {
  return (
    <BottomSheet visible={visible} onClose={onClose}>
      <View style={styles.header}>
        <Text style={styles.title}>Plant Options</Text>
        <Text style={styles.subtitle}>{plantName}</Text>
      </View>
      <ListRow
        leftIcon={<EditIcon />}
        title="Edit Plant Details"
        showChevron
        onPress={() => { onClose(); onEdit(); }}
      />
      <ListRow
        leftIcon={<HeartIcon fill={isFavorite ? '#E8997E' : 'none'} />}
        title={isFavorite ? 'Remove from Favorites' : 'Set as Favorite'}
        onPress={() => { onClose(); onToggleFavorite(); }}
      />
      <ListRow
        leftIcon={<DownloadIcon />}
        title="Export Care History"
        onPress={() => { onClose(); onExportHistory(); }}
      />
      <ListRow
        leftIcon={<ShareIcon />}
        title="Share Plant Profile"
        onPress={() => { onClose(); onShare(); }}
      />
      <ListRow
        leftIcon={<TrashIcon />}
        title="Delete Plant"
        destructive
        onPress={() => { onClose(); onDelete(); }}
      />
    </BottomSheet>
  )
}
```

## Visual Specs

- Options: standard ListRow styling
