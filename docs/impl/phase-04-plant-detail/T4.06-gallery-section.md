## Implementation
```typescript
export function GallerySection({ photos, onPhotoPress, onAddPhoto, onSeeAll }: GallerySectionProps) {
  return (
    <View>
      <SectionHeader
        title="Gallery"
        action={photos.length > 0 ? { label: 'See All', onPress: onSeeAll } : undefined}
      />
      <ScrollView horizontal showsHorizontalScrollIndicator={false} style={styles.scroll}>
        <Pressable onPress={onAddPhoto} style={styles.addButton}>
          <PlusIcon size={24} color="#5B8C5A" />
        </Pressable>
        {photos.map((photo) => (
          <Pressable key={photo.id} onPress={() => onPhotoPress(photo.id)}>
            <Image source={{ uri: photo.url }} style={styles.photo} />
          </Pressable>
        ))}
      </ScrollView>
    </View>
  )
}
```

## Visual Specs
