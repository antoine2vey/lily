## Props Interface

```typescript
  plant: {
    name: string
    species?: string
    category?: string
    health: 'healthy' | 'attention' | 'critical'
  }
}
```

## Implementation
```typescript
export function PlantHeader({ plant }: PlantHeaderProps) {
  return (
    <View style={styles.container}>
      <View style={styles.titleRow}>
        <Text style={styles.name}>{plant.name}</Text>
        <Badge
          label={plant.health.toUpperCase()}
          variant={plant.health === 'healthy' ? 'success' : 'warning'}
        />
      </View>
      {plant.species && (
        <Text style={styles.species}>{plant.category} • {plant.species}</Text>
      )}
    </View>
  )
}
```

## Visual Specs
- Name: 28px, bold, #1A1A1A
- Species/Category: 14px, regular, #5B8C5A
- Badge: positioned right of name
- Spacing: 4px between name and species
