## Implementation
```typescript
// In PlantDetailScreen
const [showDeleteConfirm, setShowDeleteConfirm] = useState(false)
const navigation = useNavigation()
const { mutate: deletePlant, isLoading: isDeleting } = useDeletePlant()

const handleDelete = () => {
  deletePlant(plant.id, {
    onSuccess: () => {
      navigation.goBack()
    },
  })
}

<ConfirmationModal
  visible={showDeleteConfirm}
  title={`Delete ${plant.name}?`}
  message="This will permanently remove all care history and photos. This action cannot be undone."
  confirmLabel="Delete Plant"
  cancelLabel="Keep Plant"
  destructive
  icon={<TrashIcon size={32} color="#E8997E" />}
  onConfirm={handleDelete}
  onCancel={() => setShowDeleteConfirm(false)}
/>
```
