## Implementation
```typescript
// Hook
export function useFertilizePlant(plantId: string) {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: async () => {
      await api.plants.fertilize(plantId)
    },
    onMutate: async () => {
      await queryClient.cancelQueries(['plant', plantId])
      const previous = queryClient.getQueryData(['plant', plantId])

      queryClient.setQueryData(['plant', plantId], (old) => ({
        ...old,
        lastFertilized: new Date(),
        daysUntilFertilize: old.fertilizingFrequency,
      }))

      return { previous }
    },
    onError: (err, variables, context) => {
      queryClient.setQueryData(['plant', plantId], context.previous)
    },
    onSettled: () => {
      queryClient.invalidateQueries(['plant', plantId])
      queryClient.invalidateQueries(['plants'])
    },
  })
}

// In PlantDetailScreen
const { mutate: fertilizePlant } = useFertilizePlant(plant.id)

const handleFertilize = () => {
  fertilizePlant()
}
```
