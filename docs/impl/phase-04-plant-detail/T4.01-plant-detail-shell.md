## Implementation
```typescript
export function PlantDetailScreen() {
  const { plantId } = useRoute<PlantDetailRoute>().params
  const { data: plant, isLoading, error } = usePlant(plantId)

  if (isLoading) return <PlantDetailSkeleton />
  if (error || !plant) return <ErrorState onRetry={refetch} />

  return (
    <View style={styles.container}>
      <ScrollView>
        <PlantHeroImage imageUrl={plant.imageUrl} />
        <View style={styles.content}>
          <PlantHeader plant={plant} />
          {/* Additional sections added in subsequent tasks */}
        </View>
      </ScrollView>
      <PlantDetailHeader plantId={plantId} />
    </View>
  )
}
```
