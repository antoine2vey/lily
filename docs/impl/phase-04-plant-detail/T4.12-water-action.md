## Implementation
```typescript
// Hook
export function useWaterPlant(plantId: string) {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: async () => {
      await api.plants.water(plantId)
    },
    onMutate: async () => {
      await queryClient.cancelQueries(['plant', plantId])
      const previous = queryClient.getQueryData(['plant', plantId])

      // Optimistic: reset water schedule
      queryClient.setQueryData(['plant', plantId], (old) => ({
        ...old,
        lastWatered: new Date(),
        daysUntilWater: old.wateringFrequency,
      }))

      return { previous }
    },
    onError: (err, variables, context) => {
      queryClient.setQueryData(['plant', plantId], context.previous)
    },
    onSettled: () => {
      queryClient.invalidateQueries(['plant', plantId])
      queryClient.invalidateQueries(['plants'])
    },
  })
}

// In PlantDetailScreen
const { mutate: waterPlant } = useWaterPlant(plant.id)

const handleWater = () => {
  waterPlant()
  // Optional: show toast "Watered!"
}
```
