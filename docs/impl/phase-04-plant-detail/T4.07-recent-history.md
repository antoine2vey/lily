## Implementation
```typescript
const EVENT_CONFIG = {
  watered: { icon: DropletIcon, color: '#2196F3', label: 'Watered' },
  fertilized: { icon: SparklesIcon, color: '#FF9800', label: 'Fertilized' },
  misted: { icon: CloudIcon, color: '#90CAF9', label: 'Mist' },
  pruned: { icon: ScissorsIcon, color: '#E91E63', label: 'Pruned' },
  repotted: { icon: FlowerIcon, color: '#9C27B0', label: 'Repotted' },
}

export function RecentHistory({ events, onViewAll }: RecentHistoryProps) {
  const recentEvents = events.slice(0, 3)

  return (
    <View>
      <SectionHeader title="Recent History" action={{ label: 'View All', onPress: onViewAll }} />
      {recentEvents.map((event) => {
        const config = EVENT_CONFIG[event.type]
        return (
          <View key={event.id} style={styles.eventRow}>
            <View style={[styles.dot, { backgroundColor: config.color }]} />
            <View style={styles.eventContent}>
              <Text style={styles.eventLabel}>{config.label}</Text>
              <Text style={styles.eventDate}>{formatDate(event.date)}</Text>
            </View>
          </View>
        )
      })}
    </View>
  )
}
```

## Visual Specs
- Event row: 48px height
- Dot: 8px circle, colored by event type
- Label: 16px, medium, #1A1A1A
- Date: 14px, regular, #757575
