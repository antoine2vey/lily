## Implementation
```typescript
const SUNLIGHT_LABELS = {
  low: 'Low Light',
  indirect: 'Indirect Bright',
  bright: 'Bright Light',
  direct: 'Direct Sun',
}

export function IdealEnvironment({ sunlight, water, humidity }: IdealEnvironmentProps) {
  return (
    <View>
      <SectionHeader title="Ideal Environment" />
      <View style={styles.rows}>
        <EnvironmentRow
          icon={<SunIcon />}
          label="Sunlight"
          value={SUNLIGHT_LABELS[sunlight]}
        />
        <EnvironmentRow
          icon={<DropletIcon />}
          label="Water"
          value={water.charAt(0).toUpperCase() + water.slice(1)}
        />
        <EnvironmentRow
          icon={<CloudIcon />}
          label="Humidity"
          value={humidity.charAt(0).toUpperCase() + humidity.slice(1)}
        />
      </View>
    </View>
  )
}

function EnvironmentRow({ icon, label, value }) {
  return (
    <View style={styles.row}>
      <View style={styles.iconContainer}>{icon}</View>
      <Text style={styles.label}>{label}</Text>
      <Badge label={value} variant="info" />
    </View>
  )
}
```

## Visual Specs
- Row height: 48px
- Icon container: 32px, light gray background
- Label: 16px, #1A1A1A
- Badge: right-aligned, info variant (light blue)
