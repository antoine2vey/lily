## Implementation
```typescript
export function useNotificationSettings() {
  const queryClient = useQueryClient()

  const query = useQuery({
    queryKey: ['notificationSettings'],
    queryFn: () => api.user.getNotificationSettings(),
  })

  const mutation = useMutation({
    mutationFn: (settings: NotificationSettings) =>
      api.user.updateNotificationSettings(settings),
    onSuccess: () => {
      queryClient.invalidateQueries(['notificationSettings'])
    },
  })

  // Debounced update
  const debouncedUpdate = useMemo(
    () => debounce((settings: NotificationSettings) => {
      mutation.mutate(settings)
    }, 500),
    []
  )

  const updateSetting = (key: string, value: any) => {
    const newSettings = { ...query.data, [key]: value }
    // Optimistic update
    queryClient.setQueryData(['notificationSettings'], newSettings)
    // Debounced API call
    debouncedUpdate(newSettings)
  }

  return {
    data: query.data,
    isLoading: query.isLoading,
    updateSetting,
    isSaving: mutation.isLoading,
  }
}
```
