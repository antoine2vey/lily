## Props Interface
```typescript
interface CareEventCardProps {
  event: {
    id: string
    type: 'watered' | 'fertilized' | 'misted' | 'pruned' | 'repotted'
    timestamp: Date
    notes?: string
    photoUrl?: string
  }
  onPress: () => void
}
```

## Implementation
```typescript
export function CareEventCard({ event, onPress }: CareEventCardProps) {
  return (
    <Pressable onPress={onPress} style={styles.card}>
      <View style={styles.header}>
        <Text style={styles.type}>{EVENT_LABELS[event.type]}</Text>
        <Text style={styles.time}>{formatTime(event.timestamp)}</Text>
      </View>
      {event.notes && (
        <Text style={styles.notes}>{event.notes}</Text>
      )}
      {event.photoUrl && (
        <Image source={{ uri: event.photoUrl }} style={styles.photo} />
      )}
    </Pressable>
  )
}
```

## Visual Specs
- Card: white, 12px border-radius, padding 12px
- Type: 16px, semibold (e.g., "Watered")
- Time: 14px, #757575 (e.g., "10:00 AM")
- Notes: 14px, #757575
- Photo: rounded, max 200px width
