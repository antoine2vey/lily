## Implementation
```typescript
interface TimelineProps {
  groups: Array<{
    date: string
    events: CareEvent[]
  }>
  onEventPress: (event: CareEvent) => void
}

export function Timeline({ groups, onEventPress }: TimelineProps) {
  return (
    <View style={styles.container}>
      {groups.map((group, groupIndex) => (
        <View key={group.date}>
          <Text style={styles.dateHeader}>{group.date}</Text>
          {group.events.map((event, eventIndex) => (
            <View key={event.id} style={styles.eventRow}>
              <View style={styles.timelineIndicator}>
                <View style={[styles.dot, { backgroundColor: getEventColor(event.type) }]} />
                {(eventIndex < group.events.length - 1 || groupIndex < groups.length - 1) && (
                  <View style={styles.line} />
                )}
              </View>
              <CareEventCard event={event} onPress={() => onEventPress(event)} />
            </View>
          ))}
        </View>
      ))}
    </View>
  )
}
```

## Visual Specs
- Dot: 12px circle, colored by event type
- Line: 2px wide, #E0E0E0, connects dots
- Date header: 18px, bold, margin top 24px
