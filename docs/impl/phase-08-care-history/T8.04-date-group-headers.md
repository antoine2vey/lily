```typescript
export function formatDateHeader(date: Date): string {
  const now = new Date()
  const today = startOfDay(now)
  const eventDate = startOfDay(date)

  if (isSameDay(eventDate, today)) {
    return 'Today'
  }

  if (isSameDay(eventDate, subDays(today, 1))) {
    return 'Yesterday'
  }

  if (isWithinInterval(eventDate, { start: subDays(today, 7), end: today })) {
    return format(date, 'EEEE') // "Wednesday"
  }

  return format(date, 'MMMM d, yyyy') // "October 20, 2023"
}

export function groupEventsByDate(events: CareEvent[]): DateGroup[] {
  const groups: Record<string, CareEvent[]> = {}

  events.forEach((event) => {
    const dateKey = startOfDay(event.timestamp).toISOString()
    if (!groups[dateKey]) {
      groups[dateKey] = []
    }
    groups[dateKey].push(event)
  })

  return Object.entries(groups)
    .sort(([a], [b]) => new Date(b).getTime() - new Date(a).getTime())
    .map(([dateKey, events]) => ({
      date: formatDateHeader(new Date(dateKey)),
      events: events.sort((a, b) => b.timestamp.getTime() - a.timestamp.getTime()),
    }))
}
```
