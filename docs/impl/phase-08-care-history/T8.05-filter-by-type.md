const FILTER_OPTIONS = [
  { key: 'all', label: 'All Events' },
  { key: 'watered', label: 'Watering' },
  { key: 'fertilized', label: 'Fertilizing' },
  { key: 'misted', label: 'Misting' },
  { key: 'pruned', label: 'Pruning' },
  { key: 'repotted', label: 'Repotting' },
]

export function FilterSheet({ visible, onClose, selectedFilter, onSelect }: FilterSheetProps) {
  return (
    <BottomSheet visible={visible} onClose={onClose} title="Filter by Type">
      {FILTER_OPTIONS.map(({ key, label }) => (
        <ListRow
          key={key}
          title={label}
          rightElement={selectedFilter === key && <CheckIcon color="#5B8C5A" />}
          onPress={() => {
            onSelect(key)
            onClose()
          }}
        />
      ))}
    </BottomSheet>
  )
}

// In CareHistoryScreen
const [filter, setFilter] = useState('all')
const filteredLogs = filter === 'all'
  ? logs
