## Implementation
```typescript
export function HomeScreen() {
  const { data: plants, isLoading } = usePlants()
  const [showAddPlant, setShowAddPlant] = useState(false)

  if (isLoading) {
    return <HomeScreenSkeleton />
  }

  if (plants?.length === 0) {
    return (
      <SafeAreaView style={styles.container}>
        <HomeHeader />
        <EmptyState
          illustration="plant"
          title="Your garden awaits"
          description="Add your first plant to start your care journey"
          action={{
            label: "Add Your First Plant",
            onPress: () => setShowAddPlant(true)
          }}
        />
        <AddPlantOptionsSheet
          visible={showAddPlant}
          onClose={() => setShowAddPlant(false)}
        />
      </SafeAreaView>
    )
  }

  // ... regular home content
}
```

## Visual Specs
- Illustration: Plant in pot image, ~200px height
- Title: 24px, semibold, centered
- Description: 16px, regular, #757575, centered
- Button: Full width with "+ Add Your First Plant"
