## Implementation
```typescript
export function HomeScreen() {
  const { data: plants, refetch: refetchPlants, isRefetching } = usePlants()
  const { refetch: refetchActivities } = useRecentActivities()
  const { refetch: refetchNotifications } = useUnreadNotificationCount()

  const onRefresh = async () => {
    await Promise.all([
      refetchPlants(),
      refetchActivities(),
      refetchNotifications(),
    ])
  }

  return (
    <SafeAreaView style={styles.container}>
      <ScrollView
        contentContainerStyle={styles.content}
        refreshControl={
          <RefreshControl
            refreshing={isRefetching}
            onRefresh={onRefresh}
            tintColor="#5B8C5A"
            colors={['#5B8C5A']} // Android
          />
        }
      >
        {/* content */}
      </ScrollView>
    </SafeAreaView>
  )
}
```
