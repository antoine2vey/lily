## Implementation
```typescript
interface ChatInputProps {
  onSend: (message: string) => void
  onAttachPhoto: () => void
  disabled?: boolean
}

export function ChatInput({ onSend, onAttachPhoto, disabled }: ChatInputProps) {
  const [text, setText] = useState('')

  const handleSend = () => {
    if (text.trim()) {
      onSend(text.trim())
      setText('')
    }
  }

  return (
    <View style={styles.container}>
      <Pressable onPress={onAttachPhoto} style={styles.attachButton}>
        <CameraIcon size={24} color="#5B8C5A" />
      </Pressable>
      <TextInput
        style={styles.input}
        value={text}
        onChangeText={setText}
        placeholder="Ask Lily about your plant..."
        placeholderTextColor="#9E9E9E"
        multiline
        maxLength={500}
      />
      <Pressable
        onPress={handleSend}
        disabled={disabled || !text.trim()}
        style={[styles.sendButton, (!text.trim() || disabled) && styles.sendButtonDisabled]}
      >
        <SendIcon size={20} color="#FFF" />
      </Pressable>
    </View>
  )
}
```
