## Implementation
```typescript
export function useSendChatMessage(plantId?: string) {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: async (input: { message: string; imageUri?: string }) => {
      let imageUrl: string | undefined
      if (input.imageUri) {
        imageUrl = await uploadChatImage(input.imageUri)
      }

      return api.chat.sendMessage({
        message: input.message,
        plantId,
        imageUrl,
      })
    },
    onMutate: async (input) => {
      // Optimistically add user message
      const userMessage = {
        id: `temp-${Date.now()}`,
        role: 'user',
        content: input.message,
        timestamp: new Date(),
      }
      queryClient.setQueryData(['chat', plantId], (old) => [...old, userMessage])
    },
    onSuccess: (response) => {
      // Add AI response
      queryClient.setQueryData(['chat', plantId], (old) => [...old, response])
    },
  })
}
```
