## Implementation
```typescript
export function useCreateCareLog() {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: async (input: CreateCareLogInput) => {
      let photoUrl: string | undefined
      if (input.photoUri) {
        photoUrl = await uploadCarePhoto(input.photoUri)
      }

      return api.careLogs.create({
        plantId: input.plantId,
        type: input.type,
        timestamp: input.timestamp,
        notes: input.notes,
        photoUrl,
      })
    },
    onSuccess: (data, variables) => {
      queryClient.invalidateQueries(['careHistory', variables.plantId])
      queryClient.invalidateQueries(['plants'])
      queryClient.invalidateQueries(['careTasks'])
    },
  })
}
```
