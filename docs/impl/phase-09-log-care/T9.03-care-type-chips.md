## Implementation
```typescript
const CARE_TYPES = [
  { key: 'watered', label: 'Watering', icon: DropletIcon },
  { key: 'fertilized', label: 'Fertilization', icon: SparklesIcon },
  { key: 'misted', label: 'Misting', icon: CloudIcon },
  { key: 'pruned', label: 'Pruning', icon: ScissorsIcon },
  { key: 'repotted', label: 'Repotting', icon: FlowerIcon },
]

export function CareTypeSelector({ value, onSelect }: CareTypeSelectorProps) {
  return (
    <View>
      <Text style={styles.label}>Care Type</Text>
      <View style={styles.chips}>
        {CARE_TYPES.map(({ key, label, icon: Icon }) => (
          <Chip
            key={key}
            label={label}
            icon={<Icon size={16} />}
            selected={value === key}
            onPress={() => onSelect(key)}
          />
        ))}
      </View>
    </View>
  )
}
```

## Visual Specs
- Chips: wrap to multiple lines
- Selected: green fill, white text
