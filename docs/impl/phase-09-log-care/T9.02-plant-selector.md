## Implementation
```typescript
interface PlantSelectorProps {
  value: string
  onSelect: (plantId: string) => void
  plants: Plant[]
}

export function PlantSelector({ value, onSelect, plants }: PlantSelectorProps) {
  const [showPicker, setShowPicker] = useState(false)
  const selectedPlant = plants.find((p) => p.id === value)

  return (
    <View>
      <Text style={styles.label}>Select Plant</Text>
      <Pressable onPress={() => setShowPicker(true)} style={styles.selector}>
        {selectedPlant ? (
          <View style={styles.selectedPlant}>
            <Image source={{ uri: selectedPlant.imageUrl }} style={styles.plantImage} />
            <Text style={styles.plantName}>{selectedPlant.name}</Text>
          </View>
        ) : (
          <Text style={styles.placeholder}>Choose a plant</Text>
        )}
        <ChevronDownIcon size={20} color="#9E9E9E" />
      </Pressable>
      <PlantPickerSheet
        visible={showPicker}
        onClose={() => setShowPicker(false)}
        plants={plants}
        onSelect={(id) => {
          onSelect(id)
          setShowPicker(false)
        }}
      />
    </View>
  )
}
```
