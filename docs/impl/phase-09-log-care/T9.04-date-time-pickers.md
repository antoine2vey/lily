## Implementation
```typescript
interface DateTimePickerProps {
  date: Date
  time: Date
  onDateChange: (date: Date) => void
  onTimeChange: (time: Date) => void
}

export function DateTimePicker({ date, time, onDateChange, onTimeChange }: DateTimePickerProps) {
  const [showDatePicker, setShowDatePicker] = useState(false)
  const [showTimePicker, setShowTimePicker] = useState(false)

  const handleNow = () => {
    const now = new Date()
    onDateChange(now)
    onTimeChange(now)
  }

  return (
    <View>
      <View style={styles.header}>
        <Text style={styles.label}>Date & Time</Text>
        <Pressable onPress={handleNow} style={styles.nowButton}>
          <Text style={styles.nowText}>NOW</Text>
        </Pressable>
      </View>
      <View style={styles.row}>
        <Pressable onPress={() => setShowDatePicker(true)} style={styles.dateButton}>
          <CalendarIcon size={20} color="#5B8C5A" />
          <Text style={styles.dateText}>{format(date, 'MM/dd/yyyy')}</Text>
        </Pressable>
        <Pressable onPress={() => setShowTimePicker(true)} style={styles.timeButton}>
          <ClockIcon size={20} color="#5B8C5A" />
          <Text style={styles.timeText}>{format(time, 'h:mm a')}</Text>
        </Pressable>
      </View>
      {/* Native DateTimePicker modals */}
    </View>
  )
}
```

## Visual Specs
